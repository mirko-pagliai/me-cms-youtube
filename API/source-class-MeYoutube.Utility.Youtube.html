29:  * An utility to get information about YouTube videos

32:  * <code>
33:  * use MeYoutube\Utility\Youtube;
34:  * </code>

36: class Youtube {
37:     /**
38:      * Parses a video url and returns the video ID
39:      * @param string $url Video url
40:      * @return mixed Video ID or FALSE
41:      */
42:     public static function getId($url) {
43:         if(preg_match('/youtube\.com/', $url)) {
44:             $url = parse_url($url);
45:             
46:             if(empty($url['query']))
47:                 return FALSE;
48:             
49:             parse_str($url['query'], $url);
50:                 
51:             return empty($url['v']) ? FALSE : $url['v'];
52:         }
53:         elseif(preg_match('/youtu.be\/(.+)$/', $url, $matches))
54:             return empty($matches[1]) ? FALSE : $matches[1];
55:         else
56:             return FALSE;
57:     }
58:     
59:     /**
60:      * Gets information about a video
61:      * @param string $id Video ID
62:      * @return mixed Array of information, otherwise FALSE
63:      * @uses MeTools\Utility\Xml::fromFile()
64:      */
65:     public static function getInfo($id) {
66:         //See https://developers.google.com/youtube/v3/getting-started#partial
67:         $url = 'https://www.googleapis.com/youtube/v3/videos?id=%s&key=%s&part=snippet,contentDetails&fields=items(snippet(title,description,thumbnails(high(url))),contentDetails(duration))';
68:         $info = Xml::fromFile(sprintf($url, $id, config('Youtube.key')));
69:                 
70:         if(empty($info['items'][0]) || empty($info['items'][0]['snippet']) || empty($info['items'][0]['contentDetails']))
71:             return FALSE;
72:                 
73:         $info = am([
74:             'preview' => $info['items'][0]['snippet']['thumbnails']['high']['url']
75:         ], $info['items'][0]['snippet'], $info['items'][0]['contentDetails']);
76:         
77:         unset($info['thumbnails']);
78:                 
79:         preg_match('/PT(([0-9]+)M)?(([0-9]+)S)?/', $info['duration'], $matches);
80:         
81:         $mins = empty($matches[2]) ? "00" : sprintf("%02d", $matches[2]);
82:         $secs = empty($matches[4]) ? "00" : sprintf("%02d", $matches[4]);
83:         
84:         $info['seconds'] = (int)$mins*60+(int)$secs;
85:         $info['duration'] = sprintf('%s:%s', $mins, $secs);
86: 
87:         return $info;
88:     }
