 39:         //The category can be passed as query string, from a widget
 40:         if($this->request->query('q'))
 41:             $this->redirect([$this->request->query('q')]);
 42:         
 43:         //Checks if the cache is valid
 44:         $this->Videos->checkIfCacheIsValid();
 45:         
 46:         //Sets the initial cache name
 47:         $cache = 'index';
 48:         
 49:         //Sets the initial conditions
 50:         $conditions = ['is_spot' => FALSE];
 51:         
 52:         //Checks if has been specified a category
 53:         if(!empty($category)) {
 54:             //Adds the category to the conditions, if it has been specified
 55:             $conditions['Categories.slug'] = $category;
 56:             
 57:             //Updates the cache name, adding the category name
 58:             $cache = sprintf('%s_%s', $cache, md5($category));
 59:         }
 60:         
 61:         //Updates the cache name with the query limit and the number of the page
 62:         $cache = sprintf('%s_limit_%s', $cache, $this->paginate['limit']);
 63:         $cache = sprintf('%s_page_%s', $cache, $this->request->query('page') ? $this->request->query('page') : 1);
 64:         
 65:         //Tries to get data from the cache
 66:         list($videos, $paging) = array_values(Cache::readMany([$cache, sprintf('%s_paging', $cache)], 'videos'));
 67:         
 68:         //If the data are not available from the cache
 69:         if(empty($videos) || empty($paging)) {
 70:             $videos = $this->paginate(
 71:                 $this->Videos->find('active')
 72:                     ->contain([
 73:                         'Categories'    => ['fields' => ['title', 'slug']],
 74:                         'Users'         => ['fields' => ['first_name', 'last_name']]
 75:                     ])
 76:                     ->select(['id', 'youtube_id', 'title', 'subtitle', 'description', 'created'])
 77:                     ->where($conditions)
 78:                     ->order([sprintf('%s.created', $this->Videos->alias()) => 'DESC'])
 79:             )->toArray();
 80:                         
 81:             //Writes on cache
 82:             Cache::writeMany([$cache => $videos, sprintf('%s_paging', $cache) => $this->request->param('paging')], 'videos');
 83:         }
 84:         //Else, sets the paging parameter
 85:         else
 86:             $this->request->params['paging'] = $paging;
 87:                 
 88:         //Sets the category title as title, if has been specified a category
 89:         if(!empty($category) && !empty($videos[0]->category->title))
 90:             $this->set('title', $videos[0]->category->title);
 91:         
 92:         $this->set(compact('videos'));
 93:     }
